#!/system/bin/sh
loga "checkmemory."
echo "nexus_memory_cache msg" > /proc/brcm/debug
loga "checkmemory.."
echo "nexus_memory msg" > /proc/brcm/debug
loga "checkmemory..."
function loga() {
/system/bin/logwrapper echo "${@}"
}
while true; do
loga "checkmemory...."
MOUNT_PATH=$(mount | grep '/storage/' | grep -v 'emulated' | awk '{print $3}')
if [ ! -z "$MOUNT_PATH" ]; then
LOG_FILE="$MOUNT_PATH/memory_info.log"
echo "========================================" >> $LOG_FILE
date >> $LOG_FILE
echo "======== /sys/kernel/debug/bhpa ========" >> $LOG_FILE
cat /sys/kernel/debug/bhpa >> $LOG_FILE
echo "=== /sys/kernel/debug/dri/0/*/client ===" >> $LOG_FILE
cat /sys/kernel/debug/dri/0/*/client >> $LOG_FILE
loga "Log write : $LOG_FILE"
else
loga "No USB."
fi
sleep 60
done
